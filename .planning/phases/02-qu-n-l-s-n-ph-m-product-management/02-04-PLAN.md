    **Part A: Add afterSave hooks for Intervention Image resize (per D-06)**

    In both `CreateProduct.php` and `EditProduct.php`, add an `afterSave()` method:

    ```php
    <?php

    namespace App\Filament\Resources\ProductResource\Pages;

    use App\Filament\Resources\ProductResource;
    use Filament\Resources\Pages\CreateRecord;
    use Illuminate\Support\Facades\Storage;
    use Intervention\Image\Laravel\Facades\Image;

    class CreateProduct extends CreateRecord
    {
        protected static string $resource = ProductResource::class;

        protected function afterSave(): void
        {
            $product = $this->getRecord();
            $images = $product->images ?? [];

            foreach ($images as $imagePath) {
                $fullPath = Storage::disk('public')->path($imagePath);
                if (file_exists($fullPath)) {
                    try {
                        $img = Image::read($fullPath)
                            ->resize(width: 600, height: 600)
                            ->encodeUsingFileExtension(
                                pathinfo($imagePath, PATHINFO_EXTENSION),
                                quality: 85
                            );
                        Storage::disk('public')->put($imagePath, $img);
                    } catch (\Exception $e) {
                        // Log resize error but don't block save
                        report($e);
                    }
                }
            }
        }
    }
    ```

    Use `Image::read()` for Intervention Image v3+ syntax (NOT `Image::decode()` which is the v2 syntax — verify the installed version via `composer show intervention/image-laravel` and use the correct API).

    Do the same for `EditProduct.php` (same `afterSave` method, same namespace).

    ```php
    <?php

    namespace Tests\Feature\Filament;

    use App\Models\User;
    use App\Models\Brand;
    use App\Models\Category;
    use App\Models\Product;
    use Illuminate\Foundation\Testing\RefreshDatabase;
    use Tests\TestCase;

    class ProductSpecsTest extends TestCase
    {
        use RefreshDatabase;

        protected function setUp(): void
        {
            parent::setUp();
            $this->actingAs(User::factory()->create([
                'email' => 'admin@example.com',
            ]));
            $this->seed([
                \Database\Seeders\CategorySeeder::class,
                \Database\Seeders\BrandSeeder::class,
            ]);
        }

        public function test_technical_specs_stores_as_array_of_objects(): void
        {
            $category = Category::whereNotNull('parent_id')->whereDoesntHave('children')->first();
            $brand = Brand::first();

            $specs = [
                ['attribute_name' => 'Công suất', 'attribute_value' => '200W'],
                ['attribute_name' => 'Điện áp', 'attribute_value' => '220V'],
            ];

            $response = $this->post('/admin/products', [
                'name' => 'Spec Test Product',
                'slug' => 'spec-test-product',
                'sku' => 'SPEC-001',
                'category_id' => $category->id,
                'brand_id' => $brand->id,
                'technical_specs' => $specs,
                'unit' => 'bộ',
                'is_active' => true,
            ]);

            $response->assertSessionHasNoErrors();

            $product = Product::where('sku', 'SPEC-001')->first();
            $this->assertNotNull($product);

            $retrievedSpecs = $product->technical_specs;
            $this->assertIsArray($retrievedSpecs);
            $this->assertCount(2, $retrievedSpecs);
            $this->assertEquals('Công suất', $retrievedSpecs[0]['attribute_name']);
            $this->assertEquals('200W', $retrievedSpecs[0]['attribute_value']);
            $this->assertEquals('Điện áp', $retrievedSpecs[1]['attribute_name']);
            $this->assertEquals('220V', $retrievedSpecs[1]['attribute_value']);
        }

        public function test_technical_specs_can_be_empty(): void
        {
            $category = Category::whereNotNull('parent_id')->whereDoesntHave('children')->first();
            $brand = Brand::first();

            $response = $this->post('/admin/products', [
                'name' => 'No Spec Product',
                'slug' => 'no-spec-product',
                'sku' => 'NOSPEC-001',
                'category_id' => $category->id,
                'brand_id' => $brand->id,
                'unit' => 'cái',
                'is_active' => true,
            ]);

            $response->assertSessionHasNoErrors();

            $product = Product::where('sku', 'NOSPEC-001')->first();
            $this->assertNotNull($product);
            $this->assertNull($product->technical_specs);
        }

        public function test_technical_specs_formats_match_frontend_blade(): void
        {
            // This test validates the format expectation that products/show.blade.php
            // now expects: @foreach($product->technical_specs as $spec)

            //   <td>{{ $spec['attribute_value'] }}</td>

            ]);

            $retrieved = $product->fresh()->technical_specs;

                $this->assertArrayHasKey('attribute_value', $spec);
            }
        }
    }
    ```
