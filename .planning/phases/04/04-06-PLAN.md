    **Sitemapable interface on models:**

    For each of Product, Category, Post, and Project models:
    1. Add import: `use Spatie\Sitemap\Contracts\Sitemapable;`
    2. Add import: `use Spatie\Sitemap\Tags\Url;`
    3. Implement the interface: `class Product extends Model implements Sitemapable`
    4. Add method:

    ```php
    public function toSitemapTag(): Url|string|array
    {
        $slug = $this->slug;
        $routeName = match (true) {
            $this instanceof \App\Models\Product => 'products.show',
            $this instanceof \App\Models\Category => 'categories.show',
            $this instanceof \App\Models\Post => 'posts.show',
            $this instanceof \App\Models\Project => 'projects.show',
            default => 'home',
        };

        return [
            Url::create(route($routeName, ['locale' => 'vi', 'slug' => $slug]))
                ->setLastModificationDate($this->updated_at ?? now())
                ->setChangeFrequency(Url::CHANGE_FREQUENCY_WEEKLY)
                ->setPriority(match (true) {
                    $this instanceof \App\Models\Product => 0.8,
                    $this instanceof \App\Models\Category => 0.7,
                    $this instanceof \App\Models\Post => 0.6,
                    $this instanceof \App\Models\Project => 0.6,
                    default => 0.5,
                })
                ->addAlternateLanguage('vi', route($routeName, ['locale' => 'vi', 'slug' => $slug]))
                ->addAlternateLanguage('en', route($routeName, ['locale' => 'en', 'slug' => $slug])),
        ];
    }
    ```

    Per D-12: Priority values are 0.8 for Products (weekly), 0.7 for Categories (weekly), 0.6 for Posts (monthly), 0.6 for Projects (monthly).

    **GenerateSitemap command:**

    Create `app/Console/Commands/GenerateSitemap.php`:

    ```php
    <?php

    namespace App\Console\Commands;

    use App\Models\Category;
    use App\Models\Post;
    use App\Models\Product;
    use App\Models\Project;
    use Illuminate\Console\Command;
    use Spatie\Sitemap\Sitemap;
    use Spatie\Sitemap\Tags\Url;

    class GenerateSitemap extends Command
    {
        protected $signature = 'sitemap:generate';
        protected $description = 'Generate the sitemap.xml with alternate locale URLs';

        public function handle()
        {
            $sitemap = Sitemap::create();

            // Static pages
            $sitemap->add(Url::create('/vi')
                ->setPriority(1.0)
                ->setChangeFrequency(Url::CHANGE_FREQUENCY_WEEKLY)
                ->addAlternateLanguage('vi', '/vi')
                ->addAlternateLanguage('en', '/en'));

            $sitemap->add(Url::create('/en')
                ->setPriority(1.0)
                ->setChangeFrequency(Url::CHANGE_FREQUENCY_WEEKLY)
                ->addAlternateLanguage('vi', '/vi')
                ->addAlternateLanguage('en', '/en'));

            $sitemap->add(Url::create('/vi/lien-he')
                ->setPriority(0.5)
                ->setChangeFrequency(Url::CHANGE_FREQUENCY_MONTHLY)
                ->addAlternateLanguage('vi', '/vi/lien-he')
                ->addAlternateLanguage('en', '/en/contact'));

            $sitemap->add(Url::create('/en/contact')
                ->setPriority(0.5)
                ->setChangeFrequency(Url::CHANGE_FREQUENCY_MONTHLY)
                ->addAlternateLanguage('vi', '/vi/lien-he')
                ->addAlternateLanguage('en', '/en/contact'));

            // Dynamic models
            Product::all()->each(fn (Product $product) => $sitemap->add($product->toSitemapTag()));
            Category::all()->each(fn (Category $category) => $sitemap->add($category->toSitemapTag()));
            Post::all()->each(fn (Post $post) => $sitemap->add($post->toSitemapTag()));
            Project::all()->each(fn (Project $project) => $sitemap->add($project->toSitemapTag()));

            $sitemap->writeToFile(public_path('sitemap.xml'));

            $this->info('Sitemap generated successfully at ' . public_path('sitemap.xml'));
        }
    }
    ```

    Note: The command uses hardcoded locale-specific URLs for static pages (home, contact) because they have different URI paths in Vietnamese vs English (/lien-he vs /contact).
